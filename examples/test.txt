fn calcPi() {
  let sum = 0;
  let d = 1;
  
  for(let i=0;i<1000000;i=i+1) {
    if(i % 2 == 0) {
      sum = sum + (4/d);  
    } else {
      sum = sum - (4/d);
    }
    d = d + 2;
  }

  print(sum);
}
