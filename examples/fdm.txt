let numVerticalSteps = 10;
let output = zeros(31, numVerticalSteps);
let pi = 3.14159265359;
let dx = 0.1;
let dt = 0.0025;

let g1 = @(x) => {
  return x * (pi - x);
};

for(let x=0;x<31;x=x+1) {
  output[0, x] = g1(x * dx);
}

for(let y=1;y<numVerticalSteps;y=y+1) {
  for(let x=1;x<30;x=x+1) {
    let c2 = ((dt)/((dx)*(dx)));
    let c1 = (1 - 2 * c2);
    output[y,x] = c1 * output[y-1, x] + c2 * output[y-1, x+1] + c2 * output[y-1, x-1];
  }
}

print(output);
